**2. Wire save in DocsView** in `packages/frontend/src/views/DocsView.tsx`:

Create `handleSave` callback:
```typescript
const handleSave = useCallback(async () => {
  if (!activeTab || !activeTab.isDirty) return;
  const editor = editorRef.current;
  if (!editor) return;

  const markdown = editor.getMarkdown();

  try {
    const res = await fetch('/api/files/save', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ filePath: activeTab.filePath, content: markdown }),
    });
    if (!res.ok) {
      const err = await res.json();
      console.error('[save] Failed:', err.error);
      return;
    }
    // Update store: save the serialized content and reset dirty flag
    updateTabContent(activeTab.id, markdown);
    setTabDirty(activeTab.id, false);
    console.log('[save] Saved:', activeTab.filePath);
  } catch (err) {
    console.error('[save] Network error:', err);
  }
}, [activeTab, updateTabContent, setTabDirty]);
```

Pass `onSave={handleSave}` to EditorToolbar.

**1. Create search extension** at `packages/frontend/src/lib/search-extension.ts`:

Wrap prosemirror-search as a TipTap extension:
```typescript
import { Extension } from '@tiptap/core';
import { search, SearchQuery, setSearchState } from 'prosemirror-search';

export interface SearchExtensionStorage {
  searchVisible: boolean;
}

export const SearchExtension = Extension.create<object, SearchExtensionStorage>({
  name: 'search',

  addStorage() {
    return { searchVisible: false };
  },

  addProseMirrorPlugins() {
    return [search()];
  },

  addCommands() {
    return {
      setSearchQuery: (query: string) => ({ tr, dispatch }) => {
        if (dispatch) {
          const sq = new SearchQuery({ search: query, caseSensitive: false });
          setSearchState(tr, sq);
          dispatch(tr);
        }
        return true;
      },
      clearSearch: () => ({ tr, dispatch }) => {
        if (dispatch) {
          setSearchState(tr, new SearchQuery({ search: '' }));
          dispatch(tr);
        }
        return true;
      },
    };
  },

  addKeyboardShortcuts() {
    return {
      'Mod-f': () => {
        this.storage.searchVisible = true;
        // The parent component reads this via editor.storage.search.searchVisible
        // and shows the SearchBar
        return true; // prevent browser's native search
      },
      'Escape': () => {
        if (this.storage.searchVisible) {
          this.storage.searchVisible = false;
          this.editor.commands.clearSearch();
          return true;
        }
        return false;
      },
    };
  },
});
```
